```c
#include "app_config.h"
#include <stdio.h>
#include <string.h>
#include "esp_log.h"

static const char *TAG = "CONFIG";

static void trim_newline(char *s)
{
    size_t len = strlen(s);
    while (len > 0 && (s[len-1] == '\n' || s[len-1] == '\r' || s[len-1] == ' ')) {
        s[--len] = '\0';
    }
}

static void trim_leading(char *s)
{
    char *p = s;
    while (*p == ' ' || *p == '\t') p++;
    if (p != s) memmove(s, p, strlen(p) + 1);
}

static esp_err_t parse_line(char *line, const char *key, char *value, size_t vmax)
{
    trim_leading(line);
    if (line[0] == '#' || line[0] == '\0') return ESP_ERR_NOT_FOUND;

    char *eq = strchr(line, '=');
    if (!eq) return ESP_ERR_NOT_FOUND;

    size_t klen = (size_t)(eq - line);
    if (klen != strlen(key)) return ESP_ERR_NOT_FOUND;
    if (strncmp(line, key, klen) != 0) return ESP_ERR_NOT_FOUND;

    const char *vstart = eq + 1;
    while (*vstart == ' ' || *vstart == '\t') vstart++;
    strncpy(value, vstart, vmax - 1);
    value[vmax - 1] = '\0';
    trim_newline(value);
    return ESP_OK;
}

esp_err_t app_config_load(app_config_t *cfg)
{
    if (!cfg) return ESP_ERR_INVALID_ARG;
    memset(cfg, 0, sizeof(*cfg));

    FILE *f = fopen("/sdcard/config.ini", "r");
    if (!f) {
        ESP_LOGW(TAG, "config.ini not found on SD card");
        return ESP_ERR_NOT_FOUND;
    }

    char line[128];
    while (fgets(line, sizeof(line), f)) {
        parse_line(line, "WIFI_SSID", cfg->wifi_ssid, sizeof(cfg->wifi_ssid));
        parse_line(line, "WIFI_PASS", cfg->wifi_pass, sizeof(cfg->wifi_pass));
        parse_line(line, "TTS_APPID", cfg->tts_appid, sizeof(cfg->tts_appid));
        parse_line(line, "TTS_TOKEN", cfg->tts_token, sizeof(cfg->tts_token));
        parse_line(line, "TTS_VOICE", cfg->tts_voice, sizeof(cfg->tts_voice));
    }
    fclose(f);

    ESP_LOGI(TAG, "Config loaded: SSID=%s, TTS_VOICE=%s",
             cfg->wifi_ssid, cfg->tts_voice);
    return ESP_OK;
}
```

```c
#include "app_wifi.h"
#include <string.h>
#include "esp_log.h"
#include "esp_wifi.h"
#include "esp_event.h"
#include "nvs_flash.h"

static const char *TAG = "WIFI";
static bool s_connected = false;
static wifi_cb_t s_callback = NULL;
static bool s_init_done = false;

static void wifi_event_handler(void *arg, esp_event_base_t base,
                               int32_t id, void *data)
{
    (void)arg;
    (void)data;

    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_START) {
        ESP_LOGI(TAG, "WiFi started, connecting...");
        esp_wifi_connect();
    } else if (base == WIFI_EVENT && id == WIFI_EVENT_STA_DISCONNECTED) {
        ESP_LOGW(TAG, "WiFi disconnected, reconnecting...");
        s_connected = false;
        if (s_callback) s_callback(false);
        esp_wifi_connect();
    } else if (base == IP_EVENT && id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t *event = (ip_event_got_ip_t *)data;
        ESP_LOGI(TAG, "Got IP: " IPSTR, IP2STR(&event->ip_info.ip));
        s_connected = true;
        if (s_callback) s_callback(true);
    }
}

esp_err_t app_wifi_init(void)
{
    if (s_init_done) return ESP_OK;

    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_netif_create_default_wifi_sta();

    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&cfg));

    ESP_ERROR_CHECK(esp_event_handler_instance_register(
        WIFI_EVENT, ESP_EVENT_ANY_ID, &wifi_event_handler, NULL, NULL));
    ESP_ERROR_CHECK(esp_event_handler_instance_register(
        IP_EVENT, IP_EVENT_STA_GOT_IP, &wifi_event_handler, NULL, NULL));

    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
    ESP_ERROR_CHECK(esp_wifi_start());

    s_init_done = true;
    ESP_LOGI(TAG, "WiFi init done");
    return ESP_OK;
}

esp_err_t app_wifi_connect(const char *ssid, const char *pass)
{
    if (!ssid || strlen(ssid) == 0) return ESP_ERR_INVALID_ARG;

    wifi_config_t wifi_cfg = {
        .sta = {
            .threshold.authmode = WIFI_AUTH_WPA2_PSK,
            .sae_pwe_h2e = WPA3_SAE_PWE_BOTH,
        },
    };
    strncpy((char *)wifi_cfg.sta.ssid, ssid, sizeof(wifi_cfg.sta.ssid) - 1);
    if (pass && strlen(pass) > 0) {
        strncpy((char *)wifi_cfg.sta.password, pass, sizeof(wifi_cfg.sta.password) - 1);
    }

    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg));
    ESP_ERROR_CHECK(esp_wifi_connect());

    ESP_LOGI(TAG, "Connecting to SSID: %s", ssid);
    return ESP_OK;
}

void app_wifi_disconnect(void)
{
    esp_wifi_disconnect();
    s_connected = false;
}

bool app_wifi_is_connected(void)
{
    return s_connected;
}

void app_wifi_set_callback(wifi_cb_t cb)
{
    s_callback = cb;
}
```

```c
#include "app_cloud_tts.h"
#include "app_config.h"
#include "board_pins.h"
#include "es8388.h"

#include <string.h>
#include <stdio.h>
#include "esp_log.h"
#include "esp_check.h"
#include "esp_http_client.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/semphr.h"
#include "driver/i2s_std.h"

static const char *TAG = "CLOUD_TTS";

/* ── Ring buffer ────────────────────────────────────────────────────── */

#define RING_BUF_SIZE  (32 * 1024)

static uint8_t *s_ring = NULL;
static volatile size_t s_ring_w = 0;
static volatile size_t s_ring_r = 0;
static volatile bool s_ring_done = false;  /* HTTP finished writing */
static SemaphoreHandle_t s_ring_mutex = NULL;

static void ring_reset(void)
{
    s_ring_w = 0;
    s_ring_r = 0;
    s_ring_done = false;
}

static size_t ring_available(void)
{
    if (!s_ring_mutex) return 0;
    xSemaphoreTake(s_ring_mutex, portMAX_DELAY);
    size_t avail = (s_ring_w >= s_ring_r)
        ? (s_ring_w - s_ring_r)
        : (RING_BUF_SIZE - s_ring_r + s_ring_w);
    xSemaphoreGive(s_ring_mutex);
    return avail;
}

static size_t ring_write(const uint8_t *data, size_t len)
{
    if (!s_ring_mutex) return 0;
    xSemaphoreTake(s_ring_mutex, portMAX_DELAY);

    size_t written = 0;
    while (written < len) {
        size_t space = (s_ring_r > s_ring_w)
            ? (s_ring_r - s_ring_w - 1)
            : (RING_BUF_SIZE - s_ring_w + (s_ring_r > 0 ? s_ring_r - 1 : RING_BUF_SIZE - 1));
        if (space == 0) break;  /* full */

        size_t chunk = len - written;
        size_t contig = RING_BUF_SIZE - s_ring_w;
        if (chunk > contig) chunk = contig;
        if (chunk > space) chunk = space;

        memcpy(s_ring + s_ring_w, data + written, chunk);
        s_ring_w = (s_ring_w + chunk) % RING_BUF_SIZE;
        written += chunk;
    }

    xSemaphoreGive(s_ring_mutex);
    return written;
}

static size_t ring_read(uint8_t *buf, size_t len)
{
    if (!s_ring_mutex) return 0;
    xSemaphoreTake(s_ring_mutex, portMAX_DELAY);

    size_t read = 0;
    while (read < len && s_ring_r != s_ring_w) {
        size_t contig = RING_BUF_SIZE - s_ring_r;
        size_t chunk = len - read;
        if (chunk > contig) chunk = contig;
        size_t avail = (s_ring_w > s_ring_r)
            ? (s_ring_w - s_ring_r)
            : (RING_BUF_SIZE - s_ring_r);
        if (chunk > avail) chunk = avail;
        if (chunk == 0) break;

        memcpy(buf + read, s_ring + s_ring_r, chunk);
        s_ring_r = (s_ring_r + chunk) % RING_BUF_SIZE;
        read += chunk;
    }

    xSemaphoreGive(s_ring_mutex);
    return read;
}

/* ── I2S handle (shared with app_tts) ───────────────────────────────── */
/* We keep our own handle, initialized once. */

static i2s_chan_handle_t s_tx_handle = NULL;
static bool s_cloud_busy = false;
static volatile bool s_stop_requested = false;
static TaskHandle_t s_playback_task = NULL;
static cloud_tts_done_cb_t s_done_cb = NULL;

/* ── HTTP data callback ─────────────────────────────────────────────── */

static esp_err_t http_data_cb(esp_http_client_event_t *evt)
{
    if (evt->event_id == HTTP_EVENT_ON_DATA) {
        if (s_stop_requested) return ESP_FAIL;
        size_t written = ring_write((uint8_t *)evt->data, evt->data_len);
        if (written < evt->data_len) {
            ESP_LOGW(TAG, "Ring buffer full, dropped %d bytes",
                     (int)(evt->data_len - written));
        }
    }
    return ESP_OK;
}

/* ── Playback task ──────────────────────────────────────────────────── */

static void playback_task(void *arg)
{
    (void)arg;

    /* Wait until ring has initial data (>50%) to avoid underrun */
    while (!s_stop_requested && ring_available() < RING_BUF_SIZE / 2) {
        if (s_ring_done) break;  /* HTTP finished with less than 50% — play anyway */
        vTaskDelay(pdMS_TO_TICKS(10));
    }

    uint8_t buf[512];
    while (!s_stop_requested) {
        size_t n = ring_read(buf, sizeof(buf));
        if (n > 0) {
            if (s_tx_handle) {
                i2s_channel_write(s_tx_handle, buf, n, NULL, pdMS_TO_TICKS(200));
            }
        } else if (s_ring_done) {
            break;  /* No more data, HTTP is done */
        } else {
            vTaskDelay(pdMS_TO_TICKS(5));
        }
    }

    /* Drain remaining */
    uint8_t drain[256];
    size_t n;
    do {
        n = ring_read(drain, sizeof(drain));
        if (n > 0 && s_tx_handle) {
            i2s_channel_write(s_tx_handle, drain, n, NULL, pdMS_TO_TICKS(200));
        }
    } while (n > 0);

    es8388_stop_playback();
    s_cloud_busy = false;

    if (s_done_cb) {
        cloud_tts_done_cb_t cb = s_done_cb;
        s_done_cb = NULL;
        cb();
    }

    vTaskDelete(NULL);
}

/* ── HTTP request ───────────────────────────────────────────────────── */

static char g_reqid[32] = "";

static void send_tts_request(const char *text)
{
    /* Generate simple reqid from tick count */
    snprintf(g_reqid, sizeof(g_reqid), "esp_%u", (unsigned)(xTaskGetTickCount()));

    char body[1024];
    int body_len = snprintf(body, sizeof(body),
        "{"
        "\"app\":{\"appid\":\"%s\",\"token\":\"%s\",\"cluster\":\"volcano_tts\"},"
        "\"user\":{\"uid\":\"esp32s3_001\"},"
        "\"audio\":{\"voice_type\":\"%s\",\"encoding\":\"pcm\","
        "\"speed_ratio\":1.0,\"volume_ratio\":1.0,\"pitch_ratio\":1.0},"
        "\"request\":{\"reqid\":\"%s\",\"text\":\"%s\",\"text_type\":\"plain\",\"operation\":\"query\"}"
        "}",
        "", /* appid — set in speak() */
        "", /* token — set in speak() */
        "", /* voice — set in speak() */
        g_reqid,
        text);

    /* Note: actual appid/token/voice filled at call site */
    /* This is a placeholder — body built in app_cloud_tts_speak() below */
    (void)body;
    (void)body_len;

    /* HTTP client config */
    esp_http_client_config_t http_cfg = {
        .url = "https://openspeech.bytedance.com/api/v1/tts",
        .method = HTTP_METHOD_POST,
        .timeout_ms = 15000,
        .buffer_size = 4096,
        .buffer_size_tx = 1024,
        .skip_cert_common_name_check = true,
        .event_handler = http_data_cb,
        .user_data = NULL,
    };

    esp_http_client_handle_t client = esp_http_client_init(&http_cfg);
    if (!client) {
        ESP_LOGE(TAG, "HTTP client init failed");
        s_cloud_busy = false;
        if (s_done_cb) {
            cloud_tts_done_cb_t cb = s_done_cb;
            s_done_cb = NULL;
            cb();
        }
        return;
    }

    /* Build body with actual credentials */
    extern app_config_t g_app_config;
    char auth_header[320];
    snprintf(auth_header, sizeof(auth_header), "Bearer %s", g_app_config.tts_token);

    char actual_body[1024];
    snprintf(actual_body, sizeof(actual_body),
        "{"
        "\"app\":{\"appid\":\"%s\",\"token\":\"%s\",\"cluster\":\"volcano_tts\"},"
        "\"user\":{\"uid\":\"esp32s3_001\"},"
        "\"audio\":{\"voice_type\":\"%s\",\"encoding\":\"pcm\","
        "\"speed_ratio\":1.0,\"volume_ratio\":1.0,\"pitch_ratio\":1.0},"
        "\"request\":{\"reqid\":\"%s\",\"text\":\"%s\",\"text_type\":\"plain\",\"operation\":\"query\"}"
        "}",
        g_app_config.tts_appid,
        g_app_config.tts_token,
        g_app_config.tts_voice,
        g_reqid,
        text);

    esp_http_client_set_header(client, "Authorization", auth_header);
    esp_http_client_set_header(client, "Content-Type", "application/json");
    esp_http_client_set_post_field(client, actual_body, strlen(actual_body));

    ESP_LOGI(TAG, "Sending TTS request (%d bytes body)", (int)strlen(actual_body));

    ring_reset();
    s_stop_requested = false;

    esp_err_t err = esp_http_client_perform(client);
    if (err == ESP_OK) {
        int status = esp_http_client_get_status_code(client);
        ESP_LOGI(TAG, "HTTP status: %d, received PCM", status);
    } else {
        ESP_LOGE(TAG, "HTTP request failed: %s", esp_err_to_name(err));
    }

    esp_http_client_cleanup(client);
    s_ring_done = true;

    /* Note: playback runs in its own task, started before HTTP */
}

/* ── Public API ─────────────────────────────────────────────────────── */

esp_err_t app_cloud_tts_init(void)
{
    if (s_tx_handle) return ESP_OK;

    /* Init I2S TX (same config as app_tts.c) */
    i2s_chan_config_t chan_cfg = {
        .id = I2S_NUM_0,
        .role = I2S_ROLE_MASTER,
        .dma_desc_num = 8,
        .dma_frame_num = 256,
        .auto_clear = true,
    };
    ESP_RETURN_ON_ERROR(i2s_new_channel(&chan_cfg, &s_tx_handle, NULL), TAG, "i2s new");

    i2s_std_config_t std_cfg = {
        .clk_cfg = I2S_STD_CLK_DEFAULT_CONFIG(TTS_SAMPLE_RATE),
        .slot_cfg = I2S_STD_PHILIPS_SLOT_DEFAULT_CONFIG(I2S_DATA_BIT_WIDTH_16BIT, I2S_SLOT_MODE_MONO),
        .gpio_cfg = {
            .mclk = PIN_I2S0_MCLK,
            .bclk = PIN_I2S0_BCLK,
            .ws   = PIN_I2S0_LRCK,
            .dout = PIN_I2S0_DOUT,
            .din  = I2S_GPIO_UNUSED,
            .invert_flags = {false, false, false},
        },
    };
    ESP_RETURN_ON_ERROR(i2s_channel_init_std_mode(s_tx_handle, &std_cfg), TAG, "i2s init");
    ESP_RETURN_ON_ERROR(i2s_channel_enable(s_tx_handle), TAG, "i2s enable");

    /* Allocate ring buffer */
    s_ring = (uint8_t *)heap_caps_malloc(RING_BUF_SIZE, MALLOC_CAP_SPIRAM | MALLOC_CAP_8BIT);
    if (!s_ring) return ESP_ERR_NO_MEM;
    memset(s_ring, 0, RING_BUF_SIZE);

    s_ring_mutex = xSemaphoreCreateMutex();
    if (!s_ring_mutex) return ESP_ERR_NO_MEM;

    ESP_LOGI(TAG, "Cloud TTS init done (I2S + ring buffer)");
    return ESP_OK;
}

esp_err_t app_cloud_tts_speak(const char *text, cloud_tts_done_cb_t cb)
{
    if (!text || strlen(text) == 0) return ESP_ERR_INVALID_ARG;

    /* Stop previous */
    if (s_cloud_busy) {
        app_cloud_tts_stop();
        vTaskDelay(pdMS_TO_TICKS(50));
    }

    s_cloud_busy = true;
    s_done_cb = cb;
    s_stop_requested = false;

    /* Sanitize text for JSON — escape quotes and backslashes */
    /* For simplicity, skip problematic chars */
    char safe_text[512];
    size_t out = 0;
    for (const char *p = text; *p && out < sizeof(safe_text) - 4; p++) {
        if (*p == '"' || *p == '\\') {
            safe_text[out++] = ' ';
        } else if ((unsigned char)*p < 0x20) {
            safe_text[out++] = ' ';
        } else {
            safe_text[out++] = *p;
        }
    }
    safe_text[out] = '\0';

    /* Start playback task first (it waits for ring buffer data) */
    xTaskCreate(playback_task, "cloud_pb", 4096, NULL, 5, NULL);

    /* Enable codec */
    es8388_start_playback();
    es8388_set_volume(-12);

    /* Send HTTP request (blocking — runs in caller's context) */
    send_tts_request(safe_text);

    return ESP_OK;
}

void app_cloud_tts_stop(void)
{
    s_stop_requested = true;
    s_ring_done = true;
    s_cloud_busy = false;
    es8388_stop_playback();
}

bool app_cloud_tts_is_busy(void)
{
    return s_cloud_busy;
```
